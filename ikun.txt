				else if ($val>=9001 and $val<=9400)

					toast <#f8e71c>鸡你太美，获得10倍奖励！只因+[$num]</#>
				else if ($val>=9401 and $val<=9700)
					$num=$base*20
					yield 1 resLcm20
					toast <#f5a623>鸡你实在是太美，获得20倍奖励！只因+[$num]</#>
				else if ($val>=9701 and $val<=9900)
					$num=$base*50
					yield 1 resLcm50

				end
			end
			yield $num bitcoins
		end
