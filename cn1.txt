	*bitcoin|bitcoins
		name:比特币
		desc:这是你拥有的比特币，你能用它们购买东西。你的目标是获得尽可能多的比特币！
		icon:icons[0,0]
		class:noBackground

	*luckycoin|lockycoins
		name:幸运币
		desc:点击花费1[n:this]，可随机获得一定量的比特币。
		icon:icons[0,0]
		class:noBackground
		show earned
		on click:
			anim icon plop
			$cost=1
			if (luckycoin<$cost)
				toast 幸运币不足!
			else
				lose $cost luckycoin
				$base=bitcoins:ps*random(55,65)
				$val=random(1,10000)
				if ($val>=1 and $val<=4000)
					$num=$base
					toast 再接再励，您获得1倍奖励。比特币+[$num]
				else if ($val>=4001 and $val<=8000)
					$num=$base*2
					yield 1 resLcm2
					toast <#0dbae6>手气不错，您获得2倍奖励。比特币+[$num]</#>
				else if ($val>=8001 and $val<=9000)
					$num=$base*5
					yield 1 resLcm5
					toast <#7ed321>啧啧称奇，您获得5倍奖励。比特币+[$num]</#>
				else if ($val>=9001 and $val<=9400)
					$num=$base*10
					yield 1 resLcm10
					toast <#f8e71c>目瞪口呆，您获得10倍奖励！比特币+[$num]</#>
				else if ($val>=9401 and $val<=9700)
					$num=$base*20
					yield 1 resLcm20
					toast <#f5a623>叹为观止，您获得20倍奖励！比特币+[$num]</#>
				else if ($val>=9701 and $val<=9900)
					$num=$base*50
					yield 1 resLcm50
					toast <#9013fe>俯首称臣，您获得50倍奖励！比特币+[$num]</#>
				else if ($val>=9901)
					$num=$base*100
					yield 1 resLcm100
					toast <#d0021b>天降伟人，您获得100倍奖励！比特币[$num]</#>
				end
			end
			yield $num bitcoins
		end
