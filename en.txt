	*bitcoin|bitcoins
		name:Bitcoins
		desc:This is the bitcoins you own, and you can use them to buy things. Your goal is to get as many bitcoins as possible!
		icon:icons[0,0]
		class:noBackground
		show earned
	*luckycoin|lockycoins
		name:Lucky Bitcoins
		desc:Click spend 1 [n:this] to get a random amount of Bitcoins. (Available from Jump Bitcoin)
		icon:icons[0,0]
		class:noBackground
		show earned
		on click:
			anim icon plop
			$cost=1
			if (luckycoin<$cost)
				toast Not enough lucky coins!
			else
				lose $cost luckycoin
				$base=bitcoins:ps*random(55,65)
				$val=random(1,10000)
				if ($val>=1 and $val<=5000)
					$num=$base*0.5
					toast <#87878b>You get [$num] bitcoins.</#>
				else if ($val>=5001 and $val<=8000)
					$num=$base
					toast Feel lucky. You get [$num] bitcoins.</#>
				else if ($val>=8001 and $val<=9600)
					$num=$base*2
					toast <#7ed321>Amazed. You get [$num] bitcoins.</#>
				else if ($val>=9601 and $val<=9890)
					$num=$base*5
					toast <#f8e71c>Stunned! You get [$num] bitcoins!</#>
				else if ($val>=9891 and $val<=9900)
					$num=$base*10
					toast <#f5a623>Breathtaking! You get [$num] bitcoins!</#>
				else if ($val>=9901 and $val<=9999)
					$num=$base*50
					toast <#9013fe>God like! You get [$num] bitcoins!</#>
				else if ($val>=10000)
					$num=$base*100
					toast <#d0021b>Legendary! You get [$num] bitcoins!</#>
				end
			end
			yield $num bitcoins
		end
